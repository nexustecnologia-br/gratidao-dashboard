        let idSelecionado = "1";

        function renderStartups() {
            const select = document.getElementById('startupSelect');
            select.innerHTML = '';
            startups.forEach(s => {
                const opt = document.createElement('option');
                opt.value = s.id;
                opt.textContent = `${s.nome} [${s.estado}]`;
                if(s.id === idSelecionado) opt.selected = true;
                select.appendChild(opt);
            });
            atualizarCardMetadados();
        }

        function atualizarCardMetadados() {
            const atual = startups.find(s => s.id === idSelecionado);
            if(atual) {
                document.getElementById('lblEstado').textContent = atual.estado;
                document.getElementById('lblCiclos').textContent = atual.ciclos + " ciclos processados";
            }
        }

        function renderChecklist() {
            const container = document.getElementById('checklistContainer');
            container.innerHTML = '';
            let concluidos = 0;

            checklistBase.forEach(item => {
                if(item.concluido) concluidos++;
                const div = document.createElement('div');
                div.className = `flex items-center justify-between p-2.5 border ${item.concluido ? 'border-success bg-zinc-900/40' : 'border-custom'}`;
                div.innerHTML = `
                    <div class="flex items-center space-x-3">
                        <input type="checkbox" ${item.concluido ? 'checked' : ''} class="w-4 h-4 accent-green-500 cursor-pointer" onchange="alternarCheck('${item.id}')">
                        <span class="${item.concluido ? 'line-through text-gray-500' : 'text-white'}">${item.item}</span>
                    </div>
                    <span class="text-[10px] px-2 py-0.5 border border-custom text-gray-400 font-bold uppercase">${item.categoria}</span>
                `;
                container.appendChild(div);
            });

            document.getElementById('progressoContagem').textContent = `${concluidos}/${checklistBase.length} OK`;
        }

        function alternarCheck(id) {
            const item = checklistBase.find(c => c.id === id);
            if(item) item.concluido = !item.concluido;
            renderChecklist();
        }

        function selecionarStartup() {
            idSelecionado = document.getElementById('startupSelect').value;
            atualizarCardMetadados();
        }

        function adicionarStartupMock() {
            const nome = document.getElementById('newNome').value.trim();
            const ideia = document.getElementById('newIdeia').value.trim();
            if(!nome || !ideia) return alert('Preencha os campos.');

            const novo = { id: Date.now().toString(), nome: nome, ideia: ideia, estado: "FASE_1_VALIDACAO", ciclos: 0 };
            startups.unshift(novo);
            idSelecionado = novo.id;
            
            document.getElementById('newNome').value = '';
            document.getElementById('newIdeia').value = '';
            
            renderStartups();
        }
